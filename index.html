            <ul>

                <li><span class="lighting">Lighting conditions</span> dramatically affect how much light is available</li>

                <li><span class="shutter-speed">Faster shutter speeds</span> reduce light but freeze motion</li>

                <li><span class="aperture">Wider apertures</span> (smaller f-numbers) increase light but reduce depth of field</li>

                <li><span class="iso">Higher ISO</span> increases light sensitivity but introduces noise</li>

            </ul>

        </div>

    <script>

        // Values for each setting

        const shutterSpeeds = ["1/8", "1/15", "1/30", "1/60", "1/125", "1/250", "1/500", "1/1000"];

        const apertures = ["f16", "f11", "f8", "f5.6", "f4", "f2.8", "f2", "f1.4"];

        const isoValues = ["100", "200", "400", "800", "1600", "3200", "6400", "12800"];

        const lightingConditions = ["Night", "Indoors", "Shade", "Overcast", "Daylight"];

        const lightingStops = [-4, -3, -2, 3, 4];

        

        // State variables

        let shutterSpeedIdx = 3;

        let apertureIdx = 3;

        let isoIdx = 2;

        let lightingIdx = 2;

        let hasSubmitted = false;

        

        // Get DOM elements

        const shutterSlider = document.getElementById('shutterSlider');

        const apertureSlider = document.getElementById('apertureSlider');

        const isoSlider = document.getElementById('isoSlider');

        const lightingSlider = document.getElementById('lightingSlider');

        const checkButton = document.getElementById('checkButton');

        const resetButton = document.getElementById('resetButton');

        const feedback = document.getElementById('feedback');

        const meterNeedle = document.getElementById('meterNeedle');

        const settingsIndicator = document.getElementById('settingsIndicator');

        

        // Calculate exposure value

        function calculateExposureValue() {

            return (7 - shutterSpeedIdx) + apertureIdx + isoIdx + lightingStops[lightingIdx];

        }

        

        // Calculate exposure stops

        function calculateExposureStops() {

            const ev = calculateExposureValue();

            const stops = ev - 10;

            return Math.min(Math.max(stops, -2), 2);

        }

        

        // Check if exposure is correct

        function isExposureCorrect() {

            const ev = calculateExposureValue();

            return ev >= 9 && ev <= 11;

        }

        

        // Generate feedback

        function generateFeedback() {

            if (isExposureCorrect()) {

                return "Perfect exposure! Your settings are well balanced.";

            } else {

                const ev = calculateExposureValue();

                const stops = calculateExposureStops();

                if (ev < 9) {

                    return `Your image would be underexposed by approximately ${Math.abs(stops).toFixed(1)} stops. Try using a slower shutter speed, wider aperture (smaller f-number), or higher ISO.`;

                } else {

                    return `Your image would be overexposed by approximately ${Math.abs(stops).toFixed(1)} stops. Try using a faster shutter speed, narrower aperture (larger f-number), or lower ISO.`;

                }

            }

        }

        

        // Update all displays

        function updateDisplays() {

            // Update slider values

            document.getElementById('shutterValue').textContent = shutterSpeeds[shutterSpeedIdx];

            document.getElementById('apertureValue').textContent = apertures[apertureIdx];

            document.getElementById('isoValue').textContent = isoValues[isoIdx];

            document.getElementById('lightingValue').textContent = lightingConditions[lightingIdx];

            document.getElementById('lightingStops').textContent = lightingStops[lightingIdx] > 0 ? `+${lightingStops[lightingIdx]} stops` : `${lightingStops[lightingIdx]} stops`;

            

            // Update display cards

            document.getElementById('displayShutter').textContent = shutterSpeeds[shutterSpeedIdx];

            document.getElementById('displayAperture').textContent = apertures[apertureIdx];

            document.getElementById('displayISO').textContent = isoValues[isoIdx];

            document.getElementById('displayLighting').textContent = lightingConditions[lightingIdx];

            document.getElementById('displayLightingStops').textContent = lightingStops[lightingIdx] > 0 ? `+${lightingStops[lightingIdx]} stops` : `${lightingStops[lightingIdx]} stops`;

            

            // Update effect descriptions

            document.getElementById('displayShutterEffect').textContent = shutterSpeedIdx < 3 ? "More light" : shutterSpeedIdx > 3 ? "Less light" : "Balanced";

            document.getElementById('displayApertureEffect').textContent = apertureIdx < 3 ? "Less light" : apertureIdx > 3 ? "More light" : "Balanced";

            document.getElementById('displayISOEffect').textContent = isoIdx < 3 ? "Less light" : isoIdx > 3 ? "More light" : "Balanced";

            

            // Update triangle

            document.getElementById('triangleLighting').textContent = lightingConditions[lightingIdx];

            

            // Update settings indicator

            const correct = isExposureCorrect();

            settingsIndicator.style.fill = hasSubmitted ? (correct ? "#049CD9" : "#F9A826") : "white";

            

            // Update light meter

            const exposureStops = calculateExposureStops();

            const needlePosition = 50 + (exposureStops * 25);

            meterNeedle.style.left = `calc(${needlePosition}% - 0.25rem)`;

            

            // Update exposure reading

            const exposureReading = document.getElementById('exposureReading');

            if (exposureStops === 0) {

                exposureReading.innerHTML = '<p>Perfect Exposure</p>';

            } else if (exposureStops < 0) {

                exposureReading.innerHTML = `<p>Underexposed by ${Math.abs(exposureStops).toFixed(1)} stops</p>`;

            } else {

                exposureReading.innerHTML = `<p>Overexposed by ${exposureStops.toFixed(1)} stops</p>`;

            }

            

            // Update feedback if submitted

            if (hasSubmitted) {

                const feedbackText = generateFeedback();

                const correct = isExposureCorrect();

                

                feedback.style.display = 'block';

                feedback.className = `feedback ${correct ? 'success' : 'failure'}`;

                document.getElementById('feedbackText').textContent = feedbackText;

                

                // Add tips for incorrect exposure

                const feedbackTips = document.getElementById('feedbackTips');

                if (!correct) {

                    let tips = `

                        <p>Try adjusting your settings to achieve a balanced exposure.</p>

                        <p>Remember: When one setting lets in more light, you may need to adjust the others to compensate.</p>

                    `;

                    

                    if (lightingIdx === 0) {

                        tips += '<p>In <strong>Night</strong> conditions, you\'ll need slower shutter speeds, wider apertures, or higher ISO to compensate for the lack of light.</p>';

                    } else if (lightingIdx === 1) {

                        tips += '<p>In <strong>Indoors</strong> conditions, you\'ll likely need slower shutter speeds or wider apertures to get enough light.</p>';

                    } else if (lightingIdx === 3) {

                        tips += '<p>In <strong>Overcast</strong> conditions, you\'ll need faster shutter speeds or narrower apertures to prevent overexposure.</p>';

                    } else if (lightingIdx === 4) {

                        tips += '<p>In bright <strong>Daylight</strong>, you\'ll need faster shutter speeds, narrower apertures, or lower ISO to avoid overexposure.</p>';

                    }

                    

                    feedbackTips.innerHTML = tips;

                } else {

                    feedbackTips.innerHTML = '';

                }

            }

        }

        

        // Event listeners

        shutterSlider.addEventListener('input', (e) => {

            shutterSpeedIdx = parseInt(e.target.value);

            updateDisplays();

        });

        

        apertureSlider.addEventListener('input', (e) => {

            apertureIdx = parseInt(e.target.value);

            updateDisplays();

        });

        

        isoSlider.addEventListener('input', (e) => {

            isoIdx = parseInt(e.target.value);

            updateDisplays();

        });

        

        lightingSlider.addEventListener('input', (e) => {

            lightingIdx = parseInt(e.target.value);

            updateDisplays();

        });

        

        checkButton.addEventListener('click', () => {

            hasSubmitted = true;

            updateDisplays();

        });

        

        resetButton.addEventListener('click', () => {

            hasSubmitted = false;

            feedback.style.display = 'none';

            shutterSpeedIdx = 3;

            apertureIdx = 3;

            isoIdx = 2;

            lightingIdx = 2;

            

            shutterSlider.value = 3;

            apertureSlider.value = 3;

            isoSlider.value = 2;

            lightingSlider.value = 2;

            

            updateDisplays();

        });

        

        // Initial update

        updateDisplays();

    </script>

</body>

</html>
